def word_count(text):
    result = {}
    for word in text.split():
        result[word] = result.get(word, 0) + 1
    return result

class OpenUpper:
    def __init__(self, filename):
        self.file = open(filename, 'r')
    def __enter__(self):
        return self.file
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.file.close()

with OpenUpper('sample.txt') as f:
    print(f.read())
