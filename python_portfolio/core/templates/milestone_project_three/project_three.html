import csv

def load_portfolio_data(file_path):
    data = []
    with open(file_path, mode='r') as file:
        reader = csv.DictReader(file)
        for row in reader:
            try:
                asset = {
                    'symbol': row['Symbol'],
                    'quantity': int(row['Quantity']),
                    'price': float(row['Price']),
                    'sector': row['Sector']
                }
                data.append(asset)
            except Exception as e:
                print(f"Skipping row due to error: {e}")
    return data

def most_expensive_asset(portfolio):
    return max(portfolio, key=lambda x: x['price'])

def group_by_sector(portfolio):
    sector_map = {}
    for asset in portfolio:
        sector = asset['sector']
        if sector not in sector_map:
            sector_map[sector] = []
        sector_map[sector].append(asset)
    return sector_map
  • ğŸ”¹ Function Used: dictionary grouping
  • ğŸ“˜ Learned: Group data items by categorical keys
  • â™»ï¸ Reusability: Can be applied to any grouping or clustering logic
  • ğŸ’¬ Comments: Efficient way to group assets by sector
