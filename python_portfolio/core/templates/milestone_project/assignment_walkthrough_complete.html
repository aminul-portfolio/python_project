# Cup Game Logic with Shuffle and Result
let cups = [' ', 'O', ' '];

function shuffleList(arr) {
  return arr
    .map(v => ({ v, r: Math.random() }))
    .sort((a, b) => a.r - b.r)
    .map(x => x.v);
}

function makeGuess(guess) {
  const shuffled = shuffleList([...cups]);
  const resultEl = document.getElementById('gameResult');
  if (shuffled[guess] === 'O') {
    resultEl.textContent = `âœ… You found it! It was under cup ${guess}.`;
    resultEl.className = 'text-success h4 mt-3';
  } else {
    resultEl.textContent = `âŒ Nope! It was under cup ${shuffled.indexOf('O')}.`;
    resultEl.className = 'text-danger h4 mt-3';
  }
}
